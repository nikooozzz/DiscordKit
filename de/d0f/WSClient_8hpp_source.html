54 void Connect(bool isReconnect = false)
55 {
56 resolver_.async_resolve(
57 gateway_url_, "443",
58 [this](const boost::system::error_code &resec, const tcp::resolver::results_type &results)
59 {
60 if (HandleError_(resec, "Resolve")) return;
61
62 asio::async_connect(ws_.next_layer().next_layer().socket(), results,
63 [this](const boost::system::error_code &conec, const tcp::endpoint &)
64 {
65 if (HandleError_(conec, "Connect")) return;
66
67 ws_.next_layer().async_handshake(
68 asio::ssl::stream_base::client,
69 [this](const boost::system::error_code &sslec)
70 {
71 if (HandleError_(sslec, "SSL Handshake")) return;
72
73 ws_.async_handshake(
74 gateway_url_, "/?v=10&encoding=json",
75 [this](const boost::system::error_code &wsec)
76 {
77 if (HandleError_(wsec, "Websocket Handshake"))
78 return;
79 SendIdentity();
80 });
81 });
82 });
83 });
84
85 std::thread([this] { ioc_.run(); }).detach();

183 std::string fetchGatewayURL_()
184 {
185 try {
186 asio::ssl::context ssl_ctx(asio::ssl::context::sslv23_client);
187
188 tcp::resolver resolver{ioc_};
189 auto const endpoint = resolver.resolve("discord.com", "https");
190
191 beast::tcp_stream stream{ioc_};
192 stream.connect(endpoint);
193
194 beast::ssl_stream<beast::tcp_stream> ssl_stream(std::move(stream), ssl_ctx);
195 ssl_stream.handshake(asio::ssl::stream_base::client);
196
197 beast::http::request<beast::http::string_body> req{beast::http::verb::get, "/api/v10/gateway", 11};
198 req.set(beast::http::field::host, "discord.com");
199 req.set(beast::http::field::user_agent, BOOST_BEAST_VERSION_STRING);
200
201 beast::http::write(ssl_stream, req);
202
203 beast::flat_buffer buffer;
204 beast::http::response<beast::http::string_body> res;
205
206 beast::http::read(ssl_stream, buffer, res);
207 std::cout << res << std::endl;

209 ssl_stream.shutdown();

211 return nlohmann::json::parse(res.body())["url"];
