52 bool handle(const Event &event)
53 {
54 if (event.opcode <= 31) {
55 return runNormalHandler(event);
56 } else if (event.opcode >= 4000 && event.opcode <= 4014) {
57 return runCloseHandler(event);
58 }
59 return false;
60 };

62 bool runNormalHandler(const Event &event)
63 {
64 if (const auto handler = handler_map_.find(event.opcode); handler != handler_map_.end()) {
65 return handler->second(event);
66 }
67
68 std::cerr << "Unknown opcode " << event.opcode << std::endl;
